CREATE proc [dbo].[cargadorOfertasAumentoCarga] (@id_Usuario int, @empresa int, @valorTarjeta decimal(18,2), @calculoAuxiliar decimal(18,2) output) as

Begin Try

	Declare @ofertaAumentoCarga int

	Declare @id_Movimiento int

	select @ofertaAumentoCarga = oferta_4AumentoCarga from Tabla_Ofertas where id_Empresa = @empresa

	if (@ofertaAumentoCarga = 1)

		Begin

			select @ofertaAumentoCarga = oferta_4ValorAumentoCarga from Tabla_ValorOfertas where id_Empresa = @empresa

			Declare @creditoUsuario decimal(18,2)
			
			select @creditoUsuario = creditoUsuario from Tabla_UsuarioCredito where id_Usuario = @id_Usuario

			update Tabla_UsuarioCredito set creditoUsuario =  @creditoUsuario + @valorTarjeta where id_Usuario = @id_Usuario

			set @calculoAuxiliar = @valorTarjeta * @ofertaAumentoCarga
		
			select @id_Movimiento = MAX(id_Movimiento) from Tabla_Movimientos

			if @id_Movimiento is null

				Begin

					set @id_Movimiento = 0

				End
	
			insert into Tabla_Movimientos (id_Movimiento, id_Usuario, id_tipoMovimiento, debe, haber, fechaMovimiento) values (@id_Movimiento + 1, @id_Usuario, 22, @valorTarjeta, null, GETDATE());  
	
			insert into Tabla_Movimientos (id_Movimiento, id_Usuario, id_TipoMovimiento, debe, haber, fechaMovimiento) values (@id_Movimiento + 2, @id_Usuario, 23, @calculoAuxiliar - @valorTarjeta, null, GetDate());
						
			select @creditoUsuario = creditoUsuario from Tabla_UsuarioCredito where id_Usuario = @id_Usuario

			update Tabla_UsuarioCredito set creditoUsuario = @creditoUsuario + @calculoAuxiliar - @valorTarjeta where id_Usuario = @id_Usuario
			
			return @calculoAuxiliar

		End

		set @calculoAuxiliar = 0

		return @calculoAuxiliar

End try

Begin Catch

	set @calculoAuxiliar = -6

	return @calculoAuxiliar

End Catch
GO

CREATE proc [dbo].[ofertaRegistro] (@id_Usuario int, @empresa int, @respuestaOferta bit output) as

	Declare @oferta_2Registro bit

	Declare @id_Movimiento int
	
	select @oferta_2Registro = oferta_2Registro from Tabla_Ofertas where id_Empresa = @empresa -- oferta al registrarse
			
	If @oferta_2Registro = 1

			select @oferta_2ValorRegistro = oferta_2ValorRegistro from Tabla_ValorOfertas where id_Empresa = @empresa 
						
			Insert Into Tabla_UsuarioCredito (id_Usuario, creditoUsuario, prestamoReal, prestamoSOS) values (@id_Usuario, @oferta_2ValorRegistro,0,0)
